(

SALARY NUMBER(5)
);

DESC STUDENTS

ALTER TABLE STUDENTS DROP COLUMN ADDRESS            //COLUMN DROP: ALTER TABLE <TABLE-NAME> DROP COLUMN <COLUMN-NAME>
ALTER TABLE STUDENTS MODIFY ID VARCHAR(12)          //COLUMN MODIFY: ALTER TABLE <TABLE-NAME> MODIFY <COLUMN-NAME> <NEW-DATATYPE> <NEW-SIZE>
ALTER TABLE STUDENTS RENAME COLUMN ID TO ROLL_NO    //COLUMN RENAME: ALTER TABLE <TABLE-NAME> RENAME COLUMN <OLD COLUMN NAME> TO <NEW COLUMN NAME>
ALTER TABLE STUDENTS RENAME TO STUDENT_DETAILS     //TABLE NAME MODIFY: ALTER TABLE <TABLE-NAME> RENAME TO <NEW-TABLE-NAME>
